مقدمة الطبعة الثانية
حــرص الاتحــاد النســائي العــام بقيــادة ســمو الشــيخة فاطمــة بنــت 
مبــارك رئيســة الاتحــاد النســائي العــام، الرئيــس الأعلــى لمؤسســة التنميــة 
الأســرية، رئيســة المجلــس الأعلــى للأمومــة والطفولــة، منــذ تأسيســه في 
 إلــى وقتنــا الحاضــر علــى العمــل فــي العديــد مــن المجــالات 1975عــام 
التــي تخــدم قطاعــات التنميــة المختلفــة، وتســهم بشــكل مباشــر 
وأساســي فــي جهــود تمكيــن المــرأة الإماراتيــة، والوصــول بهــا إلــى أعلــى 
المراتــب، والحفــاظ علــى المكتســبات التــي تــم تحقيقهــا. 
ويأتــي دوره فــي ترســيخ وتعزيــز الهويــة الوطنيــة، تلبيــة لدعــوة القيــادة 
الرشــيدة فــي الاهتمــام بالمــوروث الشــعبي وصونــه والمحافظــة عليــه 
مــن الاندثــار. بالإضافــة إلــى كونــه أحــد الأولويــات المهمــة فــي مســيرة 
عملــه، التــي اســتمرت لأكثــر مــن أربعيــن عامــً.   
وقــد شــكل اهتمــام (أم الامــارات) بهــذا الجانــب نقطــة انطــلاق للاتحــاد 
النســائي العــام لتبنــي المبــادرات والمشــاريع الراميــة إلــى الحفــاظ علــى 
المــوروث الشــعبي وصيانتــه.  وتنوعــت المبــادرات الداعمــة لجهــود 
الحفــاظ علــى الثقافــة والتــراث وترســيخ الهويــة الوطنيــة.
وتأتــي الطبعــة الثانيــة مــن كتــاب (لهجتنــا المحليــة) كجــزء أساســي 
ومهــم فــي دور الاتحــاد النســائي العــام فــي جانــب الاهتمــام بالبحــث 
العلمــي وتوثيــق التــراث بنوعيــه المــادي والمعنــوي.
وقــد حرصنــا فــي هــذه الطبعــة أن نأخــذ قــدر الإمــكان بالملاحظــات التــي 
وردت إلينــا بعــد إصــدار الطبعــة الأولــى مــن الكتــاب.  لذلــك عملنــا علــى

مــادة علميــة، إضافــة نوعيــة ومتميــزة، وأن تعمــل علــى ترســيخ موروثنــا

تحمــل لهــم الخيــر، فهــم أيضــا يهنئــون بعضهــم بعضــً عنــد نــزول

نشــكره ونثنــي عليــه.

الفصل الثالث: الثناء والهجاء

26
المــدح والــذم أو الثنــاء والهجــاء، مــن الأســاليب المســتعملة بكثــرة فــي 
الشــعر العربــي، والتــي لا تخلــو منهــا ثقافــة العديــد مــن المجتمعــات.  
فالمــدح كمــا نعلــم، يســتخدم لإبــراز الجوانــب والمواقــف الإيجابيــة فــي 
الأفــراد، كمــا أن الــذم يظهــر الصفــات الســلبية فيهــم.  وفــي ثقافتنــا لا 
نســتخدم مصطلحــات الــذم بقصــد التهكــم والســخرية، إنمــا بقصــد 
الإرشــاد والنصــح. 
	 المدح والثناء
ثمــة العديــد مــن عبــارات المــدح والثنــاء فــي لهجتنــا المحليــة، نســتعملها 
للإشــادة بالأشــخاص والإطــراء عليهــم في المواقــف والأفعال المشــرفة.  
وهــي عبــارات تحمــل دلالات واضحــة علــى إعجابنــا بهــؤلاء الأشــخاص 
واستحســاننا وتقبلنــا لمواقفهــم، ومــن هــذه العبــارات التالــي: 
-  لفــظ للمــدح والتحفيــز، بمعنــى أنــك جديــر بتحمــل كفـــو:
المســؤوليه.  وتقــال لمــن أنجــز أمــراً بشــكل تــام.  ونقــول 
للشــخص (كفــو) أو (كفــو عليــك) بغــض النظــر عــن جنســه أو 
عمــره. فهــي تقــال للرجــل والمــرأة والطفــل. 
-  هــي نــوع مــن كلمــات المــدح والإطــراء، تقــال عنــد ذكــر والنعــم:
شــخص مــا ســواء أكان حاضــراً أم غائبــً، عــرف بالرجولــة والشــهامة 
والســيره الحســنة، فتصبح كلمة (والنعم) أو (والســبع أنعــام) والتي 
يــراد بهــا التأكيــد على أنه يســتحق الثناء، مرادفً لاســمه، فمــا أن يذكر 
اســم هــذا الشــخص حتــى يقــال وراه، والنعــم والســبع أنعــام. 
-  نــوع مــن عبــارات المــدح والثنــاء، تقــال بيّــض الله وجهــك: 
لشــخص قــام بعمــل متقــن وجيــد، أو قــدم المســاعدة لأحــد، 
ممــا اســتوجب أن يكافــأ بهــذا المديــح، كمــا يمكــن أن نقــول 
(بيّــض الله وجهــك وجمّــل الله حالــك).  كمــا تقــال كنــوع مــن 
الســخرية والاســتهزاء، حيــن توجــه لشــخص لا يســتحقها.

يهــب ويبــادر لتقديــم العــون والمســاعده للاخريــن، فيقــال (فــلان 
هــب ريــح) أو (هــاب ريــح)، ويقصــد بهــا مدحــه والثنــاء عليــه لمــا

الأخــلاق الحميــدة.  كمــا نقــول للمــرأة علــى ســبيل المثــال (حافظ

إلــى خــارج القريــة وهــي تبكــي، ومــر بهــا مجموعــة مــن الأمــراء وســألوها 
عــن ســبب بكائهــا، وشــرحت لهــم معاناتهــا، فســألها كبيرهــم عــن 
اســمها وأجابــت: (شــما)، فطلــب منهــا العــودة إلــى البيــت وعليهــا أن

حيــن شــاهدوا مظهرهــم الــذي يوحــي بالإمــارة والنفــوذ، وكان هــذا
